{% highlight python %}
>>> a = 'HASSHIN!'
>>> a[::-1]
'!NIHSSAH'
>>> b = ['A', 't', 'a', 'r', 'e', '-', '!']
>>> b[::-1]
['!', '-', 'e', 'r', 'a', 't', 'A']
{% endhighlight %}
